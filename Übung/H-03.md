## 1.
$\mathrm{mul} = \mathrm{Pr}\bigl[\mathrm{const}^2_2, \mathrm{add} \circ (\mathrm{pr}^3_1,\mathrm{pr}^3_3)\bigr]$

$\mathrm{mul}(n,0) = \mathrm{const}^2_0(n,0) = 0$

$$
\begin{aligned}
\mathrm{mul}(n,m+1)
&= \mathrm{add} \circ (\mathrm{pr}^3_1,\mathrm{pr}^3_3)(n,m,\mathrm{mul}(n,m)) \\
&= \mathrm{add}\bigl(\mathrm{pr}^3_1(n,m,\mathrm{mul}(n,m)),\mathrm{pr}^3_3(n,m,\mathrm{mul}(n,m))\bigr) \\
&= \mathrm{add}\bigl(n,\mathrm{mul}(n,m)\bigr) \\
&= \mathrm{mul}(n,m) + n
\end{aligned}
$$

$\mathrm{div}(n,m)=\left\lfloor  \frac{n}{(m+1)}  \right\rfloor$
$\mathrm{div}(0,m)=0$
$\mathrm{div}(n+1,m)=\mathrm{add}(\mathrm{f_{\geq}}(n,m+1),\mathrm{div}(\mathrm{sub}(n,m+1),m))$

$$
\begin{aligned}
\mathrm{div}(n+1,m)
&= \mathrm{add}(\mathrm{f_{\geq}}(n,m+1),\mathrm{div}(\mathrm{sub}(n,m+1),m)) \\
&= \mathrm{sign}(\mathrm{sub}(n,m+1)) + \mathrm{div}(\mathrm{sub}(n,m+1),m) \\
&= \mathrm{f_{\geq}}(n,m+1) + \mathrm{div}(n-(m+1),m)
\end{aligned}
$$
